varnms <- c("plot_res", "n_sites", "sampled_area", "t_btwn_samp", 
            "study_duration")

musmu <- mus[, lapply(.SD, mean, na.rm = TRUE), by = DOI, .SDcols = varnms]

par(mfrow = c(3, 2), mar = c(3, 3, 2, 2), mgp = c(1.5, 0.5, 0))

  mu <- musmu[, get(i)]
  sdev <- vars[, get(i)]

              round(sdev / mu, 2)))
  # plot(mu, sdev / mu, xlab = "mean", ylab = "CV", main = i)
  plot(mu, sdev, xlab = "mean", ylab = "sd", main = i)
}
